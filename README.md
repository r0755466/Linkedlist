```cpp
std::string Linkedlist::to_string(void){
  // if not head
    if (!head)
        return "Linked List is empty";

    std::string output = "";

    Node *node = head;

    while (node) // Lazy evaluation!
    {
        // Never call method on nullptr
        output += node->to_string() + ""; // Output node
        node = node->next();              // Jump to next node
    }

    return output;
}
```
